002 : Hacker News Scraping

The Hacker News Scraping project is an innovative web scraper that extracts the latest technology and startup news from the Hacker News website. It demonstrates my skills in web scraping, data parsing, and staying up-to-date with the latest trends in the tech industry. This tool is ideal for tech enthusiasts and professionals who want to keep their finger on the pulse of what's happening in the world of technology and entrepreneurship.



      import requests
      from bs4 import BeautifulSoup
      import pprint



      def custom_hacker_news(links, sub):
          ln = []
          for idx, item in enumerate(links):
              title = links[idx].getText()
              href = links[idx].get('href', None)
              score = sub[idx].select('.score')
              if len(score):
                  vote = int(score[0].getText().replace(' points', ''))
                  if (vote >= 100):
                      ln.append({'title': title, 'href': href, 'vote': vote})

          return sorted(ln, key = lambda k:k['vote'], reverse=True)

      def main():
          while True:
              try:
                  p = input("which page do you want to go :")
                  if len(p) == 0:
                      break
                  p = int(p)
              except ValueError:
                  print("Enter number please!!!")
              else:
                  link = "https://news.ycombinator.com/"
                  goto = ''

                  if (p > 1):
                      goto = f"{link}/?p={str(p)}"
                  else:
                      goto = link

                  res = requests.get(goto)
                  soup = BeautifulSoup(res.text, 'html.parser')
                  links = soup.select('.titleline > a')
                  sub = soup.select('.subtext')

                  ln = custom_hacker_news(links, sub)

                  pprint.pprint(ln)



      main()
