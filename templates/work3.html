003 : Password Checker with API Integration

The Password Checker is a security-focused application that leverages API integration to ensure the strength and safety of user passwords. It checks passwords against various criteria, including length, complexity, and whether they've been compromised in data breaches. This project highlights my ability to integrate APIs, focus on user security, and develop applications that promote safe and secure online practices. It's a valuable tool for anyone looking to enhance their digital security and protect against unauthorized access.


      import requests
      import hashlib
      import re
      import sys

      def get_res(query):
          url = 'https://api.pwnedpasswords.com/range/' + query
          res = requests.get(url)
          if (res.status_code != 200):
              raise RuntimeError(f"Error fetching {res.status_code}, please check your api and query code")

          return res

      def pwned_password(password):
          sha1pass = hashlib.sha1(password.encode('utf-8')).hexdigest().upper()
          res = get_res(sha1pass[:5])
          suf = re.compile(rf'{sha1pass[5:]}')

          li = (i.rstrip().split(':') for i in res.text.split('\n'))

          #check regex of suffix and the list
          for code, count in li:
              # if hit: return response
              if (suf.fullmatch(code)):
                  return f"your Password have been pwned for {count}"
          return "Your Password is all good"

      def main(args):
          for i in args:
              print(i)
              print(pwned_password(i))

      if __name__ == '__main__':
          main(sys.argv[1:])
